  METHOD /iwbep/if_mgw_appl_srv_runtime~create_deep_entity.

    "--------------------------------------------------------------------------------------------------------------
    " IMPORTANT
    " Here are processed MASS data from OData (CREATE, UPDATE).
    " Firstly:
    " ManterProjetoSet (Project/WBS top)
    "   -Header (dummy flat structure called wrapper, but could have whatever other name)
    "   -ITEMS (dummy items, because each record has a complete project structure from OData, no needed sub-fields)
    "     Each individual record is in flat structure tomanterprojeto (do not have sub-fields)
    "     WE LOOP through tomanterprojeto to create/update each Project and top WBS.
    " ManterPepSet (Child WBS)
    "   -Header (dummy flat structure called wrapper, but could have whatever other name)
    "   -ITEMS (dummy items, because each record has a complete WBS (PEP) structure from OData, no needed of sub-fields)
    "     Each individual record is in flat structure tomanterpeps (do not have sub-fields)
    "     WE LOOP through tomanterpeps to create/update each child WBS.
    " This architecture can change in future, who knows?
    "--------------------------------------------------------------------------------------------------------------

    "Manter Projeto: Estrutura aninhada c/ Navegação p/ Manter Projetos
    DATA: BEGIN OF ls_manterprojeto.
            INCLUDE TYPE zcl_zovps_ppm_dmestres_mpc=>ts_manterprojeto.
            DATA:    tomanterprojeto TYPE STANDARD TABLE OF zcl_zovps_ppm_dmestres_mpc=>ts_manterprojetos WITH DEFAULT KEY.
    DATA: END OF ls_manterprojeto.

    "Manter PEP (children): Estrutura aninhada c/ Navegação p/ Manter PEPs
    DATA: BEGIN OF ls_manterpep.
            INCLUDE TYPE zcl_zovps_ppm_dmestres_mpc=>ts_consultarprojeto.
            DATA:    tomanterpeps TYPE STANDARD TABLE OF zcl_zovps_ppm_dmestres_mpc=>ts_manterpeps WITH DEFAULT KEY.
    DATA: END OF ls_manterpep.

*    DATA: ls_zvltps_ecosy_prj TYPE zvltps_ecosy_prj.

    DATA(lv_entityset_name) = io_tech_request_context->get_entity_set_name( ).

    DATA lt_ztabproj TYPE SORTED TABLE OF zvltps_ecosy_prj WITH NON-UNIQUE KEY proj_fase proj_ppm. " proj_sap elemento_pep.
    DATA ls_ztabproj LIKE LINE OF lt_ztabproj.
    DATA lt_ztabproj_pep TYPE SORTED TABLE OF zvltps_ecosy_prj WITH NON-UNIQUE KEY proj_fase proj_ppm proj_sap elemento_pep.
    DATA ls_ztabproj_pep LIKE LINE OF lt_ztabproj_pep.

    CONSTANTS:
      cc_success TYPE char1 VALUE 'S',
      cc_error   TYPE char1 VALUE 'E'.

    CASE lv_entityset_name.
        "--------------------------------------------------------------------------------------------------------------
        " Project/WBS top, Mantain (Create/Update)
        "--------------------------------------------------------------------------------------------------------------
      WHEN 'ManterProjetoSet'.

        "Read Odata data
        io_data_provider->read_entry_data(
        IMPORTING
          es_data = ls_manterprojeto ).

        " Select old projects
        IF ls_manterprojeto-tomanterprojeto IS NOT INITIAL.

          DATA lt_projectid_range TYPE RANGE OF zvltps_ecosy_prj-proj_ppm.
          LOOP AT ls_manterprojeto-tomanterprojeto INTO DATA(ls_item_entity).
            APPEND VALUE #( sign = 'I'
                            option = 'EQ'
                            low = ls_item_entity-projectid ) TO lt_projectid_range.
          ENDLOOP.

          SORT lt_projectid_range BY low.
          DELETE ADJACENT DUPLICATES FROM lt_projectid_range COMPARING low. " delete repeated EcoSys projects
          DELETE lt_projectid_range WHERE low IS INITIAL. "Delete empty projects

          SELECT *
            FROM zvltps_ecosy_prj
            WHERE proj_ppm IN @lt_projectid_range
             INTO TABLE @lt_ztabproj.
          IF sy-subrc <> 0.
            "it is not and error
          ENDIF.                                   "#EC EMPTY_IF_BRANCH
        ENDIF.

        " Process each project from Odata
        DATA lv_create TYPE sap_bool.
        LOOP AT ls_manterprojeto-tomanterprojeto ASSIGNING FIELD-SYMBOL(<fs_item_entity>). "#EC CI_SEL_NESTED or "#EC CI_SROFC_NESTED "#EC CI_IMUD_NESTED

          IF <fs_item_entity>-projectid IS INITIAL OR " Is key in ZControl table
             <fs_item_entity>-idfaseprojeto IS INITIAL. " Is key in ZControl table
            CONTINUE. "Current record is invalid. Go to next record.
          ENDIF.

          "Read old Zdata, if exists, to decide, if it will be to Create or to Modify the current odata project.
          CLEAR lv_create.
          CLEAR ls_ztabproj.
          READ TABLE lt_ztabproj INTO ls_ztabproj WITH TABLE KEY proj_ppm  = <fs_item_entity>-projectid
                                                                 proj_fase = <fs_item_entity>-idfaseprojeto.
          "----------------
          " CREATE
          "----------------
          IF sy-subrc <> 0.
            " this is first time this record comes in odata
            lv_create = abap_true.
          ELSEIF sy-subrc = 0 AND
         ( zcl_vlps_integ_sap_ecosys=>check_create_project( ls_ztabproj ) = abap_true OR
           zcl_vlps_integ_sap_ecosys=>check_create_pep_topo( ls_ztabproj ) = abap_true ).
            " this is second time (or more) this record comes in odata, but gave any error on last processing time.
            " so this is a retry with same record in odata of Ecosys.
            " lets give it a chance to create a project or a pep now (or both, who knows).
            lv_create = abap_true.
          ENDIF.

          IF lv_create = abap_true.

            "--------------------
            " CREATE PROJECT
            "--------------------
            IF zcl_vlps_integ_sap_ecosys=>check_create_project( ls_ztabproj ) = abap_true.

              DATA: lv_ret_create_proj_status  TYPE string,
                    lv_ret_create_proj_message TYPE string,
                    lv_ret_project             TYPE string.

              CLEAR: lv_ret_create_proj_status,
                     lv_ret_create_proj_message,
                     lv_ret_project.

              zcl_vlps_integ_sap_ecosys=>create_project(
                EXPORTING
                  ps_project    = <fs_item_entity>
                IMPORTING
                  p_ret_status  = lv_ret_create_proj_status
                  p_ret_message = lv_ret_create_proj_message
                  p_ret_project = lv_ret_project ). "#EC CI_SEL_NESTED or "#EC CI_SROFC_NESTED "#EC CALL_METH_USAGE

              IF lv_ret_create_proj_status = cc_error.
                " ERROR
                <fs_item_entity>-projectidsap     = space.
                DATA(lv_error_insert_create_proj) = zcl_vlps_integ_sap_ecosys=>insert_ztabproj(
                                                       EXPORTING
                                                         p_proj_ppm      = CONV #( <fs_item_entity>-projectid )
                                                         p_proj_fase     = CONV #( <fs_item_entity>-idfaseprojeto )
                                                         "p_chave_sap     = space "no proj-objnr yet
                                                         p_proj_sap      = space "no recording proj. for now
                                                         p_elemento_pep  = space "no PEP elem. for now
                                                         p_status        = cc_error
                                                         p_mensagem      = CONV #( lv_ret_create_proj_message ) "BAPI Error
                                                       ). "#EC OPTL_EXP

                CLEAR:
                  lv_ret_create_proj_status,
                  lv_ret_create_proj_message,
                  lv_ret_project.
                CONTINUE. " jump to next incoming project
              ELSE.

                " SUCCESS
                ls_ztabproj-proj_sap      = lv_ret_project. "created sap proj.
                "ls_ztabproj-proj_sap      = <fs_item_entity>-projectidsap = lv_ret_project. "created sap proj.
                ls_ztabproj-status        = cc_success. "BAPI Success
                ls_ztabproj-mensagem      = ''.
                lv_error_insert_create_proj = zcl_vlps_integ_sap_ecosys=>insert_ztabproj(
                                                       EXPORTING
                                                         p_proj_ppm      = CONV #( <fs_item_entity>-projectid )
                                                         p_proj_fase     = CONV #( <fs_item_entity>-idfaseprojeto )
                                                         "p_chave_sap     = space "no proj-objnr yet
                                                         p_proj_sap      = CONV #( lv_ret_project )
                                                         p_elemento_pep  = space "no PEP top yet
                                                         p_status        = cc_success
                                                         p_mensagem      = space
                                                       ). "#EC OPTL_EXP
              ENDIF.
            ENDIF.

            "----------------------------------------
            " CREATE WBS (PEP TOPO, for this proj. created above)
            "----------------------------------------
            IF zcl_vlps_integ_sap_ecosys=>check_create_pep_topo( ls_ztabproj ) = abap_true.

              DATA: lv_ret_create_pep_status  TYPE string,
                    lv_ret_create_pep_message TYPE string.

              <fs_item_entity>-projectidsap = ls_ztabproj-proj_sap.

              zcl_vlps_integ_sap_ecosys=>create_pep_topo(
                EXPORTING
                  ps_project     = <fs_item_entity>
                IMPORTING
                  p_ret_status   = lv_ret_create_pep_status
                  p_ret_message  = lv_ret_create_pep_message
                  p_elemento_pep = ls_ztabproj-elemento_pep
                  "p_chave_sap    = ls_ztabproj-chave_sap
              ).                                   "#EC CALL_METH_USAGE

              IF lv_ret_create_pep_status = cc_error.
                " ERROR
                DATA(lv_upd_pep_topo_ok) = zcl_vlps_integ_sap_ecosys=>update_ztabproj_for_pep_topo(
                                                       EXPORTING
                                                         p_proj_ppm      = CONV #( <fs_item_entity>-projectid ) "Proj PPM
                                                         p_proj_fase     = CONV #( <fs_item_entity>-idfaseprojeto ) " Proj Fase
                                                         "p_chave_sap     = space "no proj-objnr yet
                                                         p_proj_sap      = CONV #( <fs_item_entity>-projectidsap ) " Proj SAP
                                                         p_elemento_pep  = space "no PEP elem. yet
                                                         p_status        = cc_error
                                                         p_mensagem      = CONV #( lv_ret_create_pep_message ) "BAPI Error
                                                       ). "#EC OPTL_EXP

                CLEAR:
                  lv_ret_create_pep_status,
                  lv_ret_create_pep_message,
                  lv_ret_project.
                CONTINUE. " jump to next incoming project
              ELSE.

                " SUCCESS
                lv_upd_pep_topo_ok = zcl_vlps_integ_sap_ecosys=>update_ztabproj_for_pep_topo(
                                                       EXPORTING
                                                         p_proj_ppm      = CONV #( <fs_item_entity>-projectid ) "Proj PPM
                                                         p_proj_fase     = CONV #( <fs_item_entity>-idfaseprojeto ) "Proj fase
                                                         "p_chave_sap     = ls_ztabproj-chave_sap "same as proj-objnr
                                                         p_proj_sap      = CONV #( <fs_item_entity>-projectidsap ) "Proj SAP
                                                         p_elemento_pep  = ls_ztabproj-elemento_pep "same sap proj. code
                                                         p_status        = cc_success
                                                         p_mensagem      = space
                                                       ). "#EC OPTL_EXP
              ENDIF.
            ENDIF.

            "---------------
            " MODIFY PROJECT
            "---------------
            " É Quando o registro vindo do Odata
            " não existe ainda na tabela Ztab de controle
            " (é a primeira vez que recebemos este registro pelo OData vindo do EcoSys PPM)
          ELSE.

            "--------------------
            " BAPI CHANGE Project
            "--------------------
            IF ls_ztabproj-status   = cc_success AND "must be S
               ls_ztabproj-proj_sap <> ''. "must be filled

              DATA: lv_ret_upd_proj_status  TYPE string,
                    lv_ret_upd_proj_message TYPE string.

              zcl_vlps_integ_sap_ecosys=>update_project(
                EXPORTING
                  ps_project    = <fs_item_entity>
                IMPORTING
                  p_ret_status  = lv_ret_upd_proj_status
                  p_ret_message = lv_ret_upd_proj_message ). "#EC CI_SEL_NESTED or "#EC CI_SROFC_NESTED "#EC CI_IMUD_NESTED "#EC CALL_METH_USAGE


              " SUCCESS: upd ZTABLE
              IF lv_ret_upd_proj_status = cc_success.
                DATA(lv_upd_proj) = zcl_vlps_integ_sap_ecosys=>update_ztabproj(
                                                        EXPORTING
                                                          ps_project      = ls_ztabproj
                                                          p_status        = cc_success
                                                          p_mensagem      = space
                                                        ). "#EC OPTL_EXP
              ELSE.
                "ERROR, upd ZTABLE
                lv_upd_proj = zcl_vlps_integ_sap_ecosys=>update_ztabproj(
                                                       EXPORTING
                                                         ps_project      = ls_ztabproj
                                                         p_status        = cc_error
                                                         p_mensagem      = CONV #( lv_ret_upd_proj_message )
                                                       ). "#EC OPTL_EXP "#EC CI_IMUD_NESTED
                CONTINUE. "if error for current project, do not even will try pep in bellow code.
              ENDIF.

            ENDIF.

            "---------------------
            " BAPI CHANGE PEP topo
            "---------------------
            IF ls_ztabproj-status        = cc_success OR "must be S
               ls_ztabproj-elemento_pep  <> ''. "must be filled

              DATA: lv_ret_upd_pep_status  TYPE string,
                    lv_ret_upd_pep_message TYPE string.

              zcl_vlps_integ_sap_ecosys=>update_pep_topo(
                EXPORTING
                  ps_project    = <fs_item_entity>
                IMPORTING
                  p_ret_status  = lv_ret_upd_pep_status
                  p_ret_message = lv_ret_upd_pep_message ). "#EC CI_SEL_NESTED or "#EC CI_SROFC_NESTED "#EC CALL_METH_USAGE

              " SUCCESS: upd ZTABLE
              IF lv_ret_upd_pep_status = cc_success.
                DATA(lv_upd_pep) = zcl_vlps_integ_sap_ecosys=>update_ztabproj(
                                                       EXPORTING
                                                         ps_project      = ls_ztabproj
                                                         p_status        = cc_success
                                                         p_mensagem      = space
                                                       ). "#EC OPTL_EXP
              ELSE.
                "ERROR, upd ZTABLE
                lv_upd_pep = zcl_vlps_integ_sap_ecosys=>update_ztabproj(
                                                       EXPORTING
                                                         ps_project      = ls_ztabproj
                                                         p_status        = cc_error
                                                         p_mensagem      = CONV #( lv_ret_upd_pep_message )
                                                       ). "#EC OPTL_EXP
              ENDIF.

            ENDIF.

          ENDIF.

        ENDLOOP.

        copy_data_to_ref(
        EXPORTING
          is_data = ls_manterprojeto
        CHANGING
          cr_data = er_deep_entity ).

        "--------------------------------------------------------------------------------------------------------------
        " Manter PEP child (WBS child, Create/Modify)
        "--------------------------------------------------------------------------------------------------------------
      WHEN 'ManterPepSet'.

        "Read Odata data
        io_data_provider->read_entry_data(
        IMPORTING
          es_data = ls_manterpep ).

        "Error. No data to process
        IF ls_manterpep-tomanterpeps IS INITIAL.
          RAISE EXCEPTION TYPE /iwbep/cx_mgw_busi_exception
            EXPORTING
              textid  = /iwbep/cx_mgw_busi_exception=>business_error
              message = 'Error. No data to process'(003).
        ENDIF.

        "ManterPep(1,header dummy) --> to MangerPeps(N, records)
        IF ls_manterpep-tomanterpeps IS NOT INITIAL.

          "Prepare RANGES for select old peps (if they exist)
          DATA lt_proj_ppm_range TYPE RANGE OF zvltps_ecosy_prj-proj_ppm.
          DATA lt_proj_fase_range TYPE RANGE OF zvltps_ecosy_prj-proj_fase.
          DATA lt_proj_sap_range TYPE RANGE OF zvltps_ecosy_prj-proj_sap.
          DATA lt_elemento_pep_range TYPE RANGE OF zvltps_ecosy_prj-elemento_pep.
          LOOP AT ls_manterpep-tomanterpeps INTO DATA(ls_tomanterpep).
            "Fase proj PPM
            APPEND VALUE #( sign = 'I'
                            option = 'EQ'
                            low = ls_tomanterpep-idfaseprojeto ) TO lt_proj_fase_range.

            "proj PPM
            APPEND VALUE #( sign = 'I'
                            option = 'EQ'
                            low = ls_tomanterpep-projectid ) TO lt_proj_ppm_range.
            "proj SAP
            APPEND VALUE #( sign = 'I'
                            option = 'EQ'
                            low = ls_tomanterpep-projectidsap ) TO lt_proj_sap_range.
            "Elem. PEP
            APPEND VALUE #( sign = 'I'
                            option = 'EQ'
                            low = ls_tomanterpep-elementopep ) TO lt_elemento_pep_range.
          ENDLOOP.

          "Read Z-records, in case they exist.
          lt_ztabproj_pep = zcl_vlps_integ_sap_ecosys=>read_zpep_by_range(
                                                              ir_proj_ppm  = lt_proj_ppm_range
                                                              ir_proj_fase = lt_proj_fase_range
                                                              ir_proj_sap = lt_proj_sap_range
                                                              ir_elemento_pep = lt_elemento_pep_range
                                                            ).
        ENDIF.

        " Process each project from Odata
        DATA lv_create_pep TYPE sap_bool.
        LOOP AT ls_manterpep-tomanterpeps ASSIGNING FIELD-SYMBOL(<fs_tomanterpeps>). "#EC CI_SEL_NESTED or "#EC CI_SROFC_NESTED "#EC CI_IMUD_NESTED

          "Check if current ODATA RECORD is valid as a child pep
          IF <fs_tomanterpeps>-projectid IS INITIAL OR
             <fs_tomanterpeps>-idfaseprojeto IS INITIAL OR
             <fs_tomanterpeps>-projectidsap IS INITIAL OR
             <fs_tomanterpeps>-elementopep IS INITIAL.
            CONTINUE. "Error: Ignore Current record is invalid. Go to next.
          ENDIF.

          "Check if current child pep is valid as child pep
          IF zcl_vlps_integ_sap_ecosys=>check_create_pep_filho( iv_pep_child = <fs_tomanterpeps>-elementopep ) = abap_false.
            CONTINUE. "Error: Ignore Current PEP is invalid as child pep. Go to next.
          ENDIF.

          "Read old Z-data, if exists, to decide, if it will be to Create or to Modify the current odata project.
          CLEAR lv_create_pep.
          CLEAR ls_ztabproj_pep.
          READ TABLE lt_ztabproj_pep INTO ls_ztabproj_pep WITH TABLE KEY proj_fase    = <fs_tomanterpeps>-idfaseprojeto
                                                                         proj_ppm     = <fs_tomanterpeps>-projectid
                                                                         proj_sap     = <fs_tomanterpeps>-projectidsap
                                                                         elemento_pep = <fs_tomanterpeps>-elementopep.
          IF sy-subrc <> 0.
            lv_create_pep = abap_true.
          ELSEIF sy-subrc = 0 AND
                 zcl_vlps_integ_sap_ecosys=>check_create_pep_filho_bapi( ls_ztabproj_pep ) = abap_true.
            lv_create_pep = abap_true.
          ELSE.
            lv_create_pep = abap_false.
          ENDIF.

          "-----------------------------
          " CREATE pep filho (WBS child)
          "-----------------------------
          IF lv_create_pep = abap_true.

            DATA: lv_chave_sap TYPE zvltps_ecosy_prj-chave_sap.

            zcl_vlps_integ_sap_ecosys=>create_pep(
              EXPORTING
                ps_pep        = <fs_tomanterpeps>
              IMPORTING
                p_ret_status  = lv_ret_create_pep_status
                p_ret_message = lv_ret_create_pep_message
                "p_chave_sap   = lv_chave_sap "ls_ztabproj-chave_sap "same as proj-objnr
            ).

            IF lv_ret_create_pep_status = cc_error.
              " ERROR
              lv_upd_pep = zcl_vlps_integ_sap_ecosys=>insert_ztabproj_pep_child(
                                                       EXPORTING
                                                         p_proj_ppm      = CONV #( <fs_tomanterpeps>-projectid ) "Proj PPM
                                                         p_proj_fase     = CONV #( <fs_tomanterpeps>-idfaseprojeto ) " Proj Fase
                                                         "p_chave_sap     = space "no proj-objnr yet
                                                         p_proj_sap      = CONV #( <fs_tomanterpeps>-projectidsap ) " Proj SAP
                                                         p_elemento_pep  = CONV #( <fs_tomanterpeps>-elementopep )
                                                         p_status        = cc_error
                                                         p_mensagem      = CONV #( lv_ret_create_pep_message ) "BAPI Error
                                                       ). "#EC OPTL_EXP
              CONTINUE. " jump to next incoming project
            ELSE.
              " SUCCESS
              lv_upd_pep = zcl_vlps_integ_sap_ecosys=>insert_ztabproj_pep_child(
                                                       EXPORTING
                                                         p_proj_ppm      = CONV #( <fs_tomanterpeps>-projectid ) "Proj PPM
                                                         p_proj_fase     = CONV #( <fs_tomanterpeps>-idfaseprojeto ) "Proj fase
                                                         "p_chave_sap     = ls_ztabproj-chave_sap "same as proj-objnr
                                                         p_proj_sap      = CONV #( <fs_tomanterpeps>-projectidsap ) "Proj SAP
                                                         p_elemento_pep  = CONV #( <fs_tomanterpeps>-elementopep )
                                                         p_status        = cc_success
                                                         p_mensagem      = space
                                                       ). "#EC OPTL_EXP
            ENDIF.

            "-----------------------------
            " MODIFY pep filho (WBS child)
            "-----------------------------
          ELSE.

            zcl_vlps_integ_sap_ecosys=>update_pep(
              EXPORTING
                ps_pep        = <fs_tomanterpeps>
              IMPORTING
                p_ret_status  = lv_ret_upd_pep_status
                p_ret_message = lv_ret_upd_pep_message
            ).

            " ERROR
            IF lv_ret_upd_pep_status = cc_error.
              lv_upd_pep = zcl_vlps_integ_sap_ecosys=>update_ztabproj_for_pep_child(
                                                     EXPORTING
                                                       ps_project      = ls_ztabproj_pep
                                                       p_status        = cc_error
                                                       p_mensagem      = CONV #( lv_ret_upd_pep_message )
                                                       ).
              CONTINUE. " jump to next incoming project

            ELSE.
              " SUCCESS
              lv_upd_pep = zcl_vlps_integ_sap_ecosys=>update_ztabproj_for_pep_child(
                                                     EXPORTING
                                                       ps_project      = ls_ztabproj_pep
                                                       p_status        = cc_success
                                                       p_mensagem      = space
                                                     ).   "#EC OPTL_EXP
            ENDIF.
          ENDIF.
        ENDLOOP.

        "return data
        copy_data_to_ref(
        EXPORTING
          is_data = ls_manterpep
        CHANGING
          cr_data = er_deep_entity ).

    ENDCASE.

  ENDMETHOD.
